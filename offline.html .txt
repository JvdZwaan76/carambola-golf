    <script>
        // Check connection status
        function updateConnectionStatus() {
            const statusEl = document.getElementById('connectionStatus');
            if (navigator.onLine) {
                statusEl.className = 'connection-status online';
                statusEl.innerHTML = '‚úÖ Connection restored!';
                setTimeout(() => {
                    window.location.reload();
                }, 2000);
            } else {
                statusEl.className = 'connection-status offline';
                statusEl.innerHTML = '‚ùå You\'re currently offline';
            }
        }

        function checkConnection() {
            updateConnectionStatus();
            
            // Try to fetch a small resource to verify connectivity
            if (navigator.onLine) {
                fetch('/favicon.ico', { 
                    mode: 'no-cors',
                    cache: 'no-cache'
                }).then(() => {
                    window.location.reload();
                }).catch(() => {
                    const statusEl = document.getElementById('connectionStatus');
                    statusEl.className = 'connection-status offline';
                    statusEl.innerHTML = '‚ùå Still having connection issues';
                });
            }
        }

        // Listen for connection changes
        window.addEventListener('online', updateConnectionStatus);
        window.addEventListener('offline', updateConnectionStatus);

        // Initial status check
        updateConnectionStatus();

        // Periodic connection check
        setInterval(() => {
            if (navigator.onLine) {
                checkConnection();
            }
        }, 5000);

        // Service Worker messaging
        if ('serviceWorker' in navigator) {
            navigator.serviceWorker.addEventListener('message', event => {
                if (event.data && event.data.type === 'CACHE_UPDATED') {
                    const statusEl = document.getElementById('connectionStatus');
                    statusEl.className = 'connection-status online';
                    statusEl.innerHTML = '‚úÖ Content updated! Refreshing...';
                    setTimeout(() => window.location.reload(), 1000);
                }
            });
        }

        console.log('üèåÔ∏è Carambola Golf Club - Offline Mode');
        console.log('üì± This page works offline thanks to service workers');
    </script>
